//Program to calculate the square of a number//
#include <stdio.h>

int main()
{
    int n, s;
    printf("Enter a number: ");
    scanf("%d", &n);
    s = n * n;
    printf("The square of %d" " is: %d", n,s);

    return 0;
}





OUTPUT:
Enter a number: 6
The square of 6 is: 36
