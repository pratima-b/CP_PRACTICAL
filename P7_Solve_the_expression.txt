//Solve the expression (a/b) + (c/d)//

#include <stdio.h>

int main()
{
    int a,b,c,d;
    float e;
    printf("Enter the value of a: ");
    scanf("%d", &a);
    printf("Enter the value of b: ");
    scanf("%d", &b);
    printf("Enter the value of c: ");
    scanf("%d", &c);
    printf("Enter the value of d: ");
    scanf("%d", &d);
    e = ((float)a/b) + ((float)c/d);
    printf("Answer of the expression (%d/%d) + (%d/%d) is: %f", a,b,c,d,e);
    
    return 0;
}





OUTPUT:
Enter the value of a: 1
Enter the value of b: 2
Enter the value of c: 1
Enter the value of d: 2
Answer of the expression (1/2) + (1/2) is: 1.000000
